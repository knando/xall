Código validando se um nome e sobrenome foram digitados

			
			algoritmo "Enquanto, contador e flag"

			var
			  nome: caractere
			  contador: inteiro
			  // Flag significa bandeira. Neste contexto é como se levantassemos uma bandeira indicando algo - similar ao que é feito em esportes.
			  flagNomeEhValido: logico

			inicio
			  // ENQUANTO o texto tiver menos que 1 caractere OU não encontrar um ESPAÇO (ou seja, não foi digitado nome E sobrenome)
			  enquanto nao flagNomeEhValido e contador < 3 faca
				escreval("Digite seu nome completo: ")
				leia(nome)

				// Ao adicionar o resultado de um teste a uma variável lógica, tornamos possível reutilizar este resultado
				flagNomeEhValido <- Compr(nome) > 1 e Pos(nome, " ") <> 0

				se nao flagNomeEhValido entao
				  contador <- contador + 1
				  escreval("Erro: Você não digitou um nome e sobrenome.")
				fimse
			  fimenquanto
			  limpatela
			  // Utilizamos a flag para indicar que, ao fim da execução, não atingimos o resultado esperado.
			  se flagNomeEhValido entao
				escreval("Ocorreram ", contador, " erros e o nome é ", nome)
			  senao
				escreval("O usuário não digitou um nome válido e excedeu as tentativas.")
			  fimse
